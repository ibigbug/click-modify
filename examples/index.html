js脚本如下

seajs.use('../src/click-modify', function(ClickModify){
        var options = {
            trigger: '.click-modify',
            type: 'input',
        };
        var click_modify = new ClickModify(options, function($trigger, $box){
            console.log($trigger);
            console.log($box);
            }));
        });
